import sys
input = sys.stdin.readline
from collections import deque


def get_card(num):
    queue = deque([n for n in range(1, num+1)])
    while len(queue) >1:
        queue.popleft()
        queue.append(queue.popleft())
        
    return queue.pop()


if __name__ == '__main__':
    n = int(input())
    print(get_card(n))
