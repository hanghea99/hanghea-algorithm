```js
function fun(arr) {
  const stack = [];

  for (let i = 0; i < arr.length; i++) {
    if (arr[i] === "(") {
      stack.push("(");
    } else{
        if(stack.length === 0) return "NO";
        if(stack.pop() !=="(") return "NO";
    } 
  }
  return stack.length ? "NO" : "YES";
};
```

### py code
```py
num = int(input())

for i in range(num):
  input_data = input()
  stack = []

  for j in input_data:
    if j == "(":
      stack.append(j)
    elif j == ")":
      if len(stack) != 0 and stack[-1] == "(":
        stack.pop()
      else:
        stack.append(")")
        break

  if len(stack) == 0:
    print("YES")
  else:
    print("NO")
```
