from collections import deque
import sys
input = sys.stdin.readline

def print_que():
    n,m=map(int,input().split( ))
    imp=list(map(int,input().split( )))
    

    queue = deque(imp)
    count=0
    while queue:
        best = max(queue)
        front = queue.popleft()
        m-=1
        
        if best == front:
            count +=1
            if m<0:
                print(count)
                break
        else:
            queue.append(front)
            if m<0:
                m=len(queue)-1
if __name__ == '__main__':
    n = int(input())
    # numArr = [int(input()) for _ in range(n)]
    for _ in range(n):
        print_que()
